1.  (2.1.0)

jgem install rails --include-dependencies --no-rdoc --no-ri
jgem install jruby-openssl --no-rdoc --no-ri
jgem update --system
jruby --version
jgem --version
jrails --version

2.

jrails --database mysql GroomingSalon

3.
cd GroomingSalon

4.

start jruby script/server

5.
jruby script/generate controller homes

6. 

jruby script\generate scaffold Cat
jruby script\generate scaffold Dog

jruby script\generate scaffold Pet veterinar:string referredBy:string medicalProblems:text breed:string size:string name:string sex:string color:string birthDate:datetime clip1:string clip2:string clip3:string specialInstructions:text behavior:text
jruby script\generate scaffold PetOwner firstName:string lastName:string homePhone:string workPhone:string cellPhone:string salutation:string

jruby script\generate scaffold Appointment appointmentDate:datetime price:string

jruby script\generate scaffold Groomer firstName:string lastName:string notes:text


7.

jrake db:drop:all 
jrake db:create:all 
jrake db:migrate 
jrake test

jrake db:version


8. Open your browser to 

http://localhost:3000/cats

jruby script/generate migration add_pet_owner_id_to_pets
